CREATE TABLE usuario(idUsuario serial primary key AUTO_INCREMENT,

create table endereco(idEndereco int not null primary key AUTO_INCREMENT,

			logradouro varchar(255) not null,
            
			complemento varchar(255) null,
            
			bairro varchar(255) not null,
            
			numero int not null,
            
			CEP int not null
            
 
)
